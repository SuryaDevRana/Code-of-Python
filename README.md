# Code-of-Python

# Variables and Data Types:
# Variables
name = "John"
age = 25
height = 5.9

# Data Types
print(type(name))   # <class 'str'>
print(type(age))    # <class 'int'>
print(type(height)) # <class 'float'>

a = 10
b = 5

# Addition
sum_result = a + b
print("Sum:", sum_result)

# Subtraction
difference = a - b
print("Difference:", difference)

# Multiplication
product = a * b
print("Product:", product)

# Division
quotient = a / b
print("Quotient:", quotient)
